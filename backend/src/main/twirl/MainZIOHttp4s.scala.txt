@(groupId: String)
package @groupId

import org.http4s.blaze.server.BlazeServerBuilder
import org.http4s.server.Router
import sttp.tapir.server.http4s.ztapir.ZHttp4sServerInterpreter
import zio.interop.catz._
import zio.{Scope, Task, ZIO, ZIOAppArgs, ZIOAppDefault}

object Main extends ZIOAppDefault {

  override def run: ZIO[Any with ZIOAppArgs with Scope, Any, Any] = {

    val routes = ZHttp4sServerInterpreter().from(List(ApiDefinitions.helloServerEndpoint)).toRoutes

    val serve: Task[Unit] =
      BlazeServerBuilder[Task]
        .withExecutionContext(runtime.runtimeConfig.executor.asExecutionContext)
        .bindHttp(8080, "localhost")
        .withHttpApp(Router("/" -> (routes)).orNotFound)
        .serve
        .compile
        .drain

    serve.exitCode

  }
}
